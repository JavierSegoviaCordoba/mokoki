 public inline fun <reified T : Any> log(
     priority: Priority = DEBUG,
     tag: String? = null,
     message: () -> T
 ) {
     for (logger in MokokiLogger.loggers) {
         if (logger.isLoggable(priority)) logger.log(priority, tag, T::class, typeOf<T>(), message())
     }
 }
